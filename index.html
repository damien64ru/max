const express = require('express'); const axios = require('axios'); const fs = require('fs'); const path = require('path'); const { initDB, getDb } = require('./db'); const { headers, BASE_URL } = require('./config'); const { sendMessage } = require('./helpers/messages'); const { mainMenuButtons } = require('./handlers/menu'); const { processUpdate } = require('./handlers/webhook'); // Гарантируем создание папки для временных файлов const TEMP_DIR = '/data/temp'; if (!fs.existsSync(TEMP_DIR)) { fs.mkdirSync(TEMP_DIR, { recursive: true }); console.log(`✅ Директория для временных файлов создана: ${TEMP_DIR}`); } else { console.log(`✅ Директория для временных файлов уже существует: ${TEMP_DIR}`); } const app = express(); app.use(express.json()); // CORS app.use((req, res, next) => { res.header('Access-Control-Allow-Origin', '*'); res.header('Access-Control-Allow-Headers', 'Origin, X-Requested-With, Content-Type, Accept'); res.header('Access-Control-Allow-Methods', 'GET, POST, OPTIONS'); if (req.method === 'OPTIONS') return res.sendStatus(200); next(); }); // Маршрут для мини-приложения app.get('/getLink', async (req, res) => { const campaignCode = req.query.campaignId; const userId = req.query.userId; console.log(`🔗 Запрос /getLink: campaignId=${campaignCode}, userId=${userId}`); if (!campaignCode) return res.status(400).json({ error: 'Missing campaignId' }); const db = getDb(); const campaign = await db.get(`SELECT * FROM campaigns WHERE tracking_link LIKE ?`, `%${campaignCode}%`); if (!campaign) return res.status(404).json({ error: 'Campaign not found' }); await db.run(`UPDATE campaigns SET clicks_count = clicks_count + 1 WHERE id = ?`, campaign.id); if (userId) { const { isUserAdminOfChannel, isUserMemberOfChannel } = require('./utils/common'); const isAdmin = await isUserAdminOfChannel(userId, campaign.channel_id); if (isAdmin) { await db.run(`INSERT INTO clicks (campaign_id, user_id, clicked_at, attended) VALUES (?, ?, ?, 1)`, campaign.id, userId, Date.now()); console.log(`👤 Администратор ${userId} перешёл по ссылке, не добавляем в ожидание`); } else { const alreadyMember = await isUserMemberOfChannel(userId, campaign.channel_id); if (alreadyMember) { await db.run(`INSERT INTO clicks (campaign_id, user_id, clicked_at, attended) VALUES (?, ?, ?, 1)`, campaign.id, userId, Date.now()); console.log(`👤 Пользователь ${userId} уже подписан, клик засчитан, но подписка не ожидается`); } else { await db.run(`INSERT INTO clicks (campaign_id, user_id, clicked_at, attended) VALUES (?, ?, ?, 0)`, campaign.id, userId, Date.now()); console.log(`👤 Пользователь ${userId} перешёл по ссылке кампании ${campaign.name} (ожидает подписки)`); } } } else { console.log(`⚠️ Запрос без userId для кампании ${campaign.name}`); } res.json({ link: campaign.channel_link }); }); app.post('/webhook', async (req, res) => { try { await processUpdate(req.body); res.sendStatus(200); } catch (err) { console.error("Ошибка обработки webhook:", err); res.sendStatus(500); } }); async function setWebhook() { const webhookUrl = `https://maks-damien64ru.amvera.io/webhook`; try { await axios.post(`${BASE_URL}/subscriptions`, { url: webhookUrl }, { headers }); console.log(`✅ Webhook установлен на ${webhookUrl}`); } catch (err) { console.error("Ошибка установки webhook:", err.response?.data || err.message); } } async function start() { await initDB(); const PORT = process.env.PORT || 3000; app.listen(PORT, async () => { console.log(`🌐 HTTP-сервер на порту ${PORT}`); await setWebhook(); }); } start();
